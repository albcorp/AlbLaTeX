\documentclass[10pt,a4paper,twoside,reqno]{alb-thesis}

%
% Encapsulated postscript handling
%
% See ``grfguide.pdf``

\usepackage{graphicx}

%
% Author Year Citations
%
% See ``natbib.dvi``

\usepackage[numbers]{natbib}

% Increase the hanging indent
\setlength{\bibhang}{2em}

%
% Cross referencing in hypertext environments
%
% This should be the one of the last packages loaded
% See ``hyperref/manual.pdf``

\usepackage[
    colorlinks=true, raiselinks=true, plainpages=false,
    bookmarks=true, bookmarksopen=true, bookmarksnumbered=true,
    hyperindex=true, pdftex
  ]{hyperref}

%
% Indices
%
% Use the LaTeX glossary and index systems to generate the raw notation
% and general indices respectively  The raw indices are processed with
% MakeIndex

% Define the \printindex command to read the files generated by MakeIndex.
\makeatletter
\newcommand{\printindex}[0]{\@input@{\jobname.ind}}
\makeatother

% Redefine theindex environment without a default heading.
\makeatletter
\renewenvironment{theindex}[0]{%
  \begin{multicols}{2}%
    \let\item\@idxitem%
}{%
  \end{multicols}%
}
\makeatother

% Make a general index
\makeindex
